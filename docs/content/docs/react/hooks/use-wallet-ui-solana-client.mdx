import { Callout } from 'fumadocs-ui/components/callout';

The `useWalletUiSolanaClient` hook provides direct access to the `SolanaClient` instance from the `gill` library.

This client is automatically configured to point to the RPC endpoint of the currently selected cluster.

<Callout>
    For most common use cases, you may not need to use this hook directly. The `useWalletUi` hook already provides
    access to the client, and other hooks like `useWalletUiSigner` use it internally. This hook is for advanced use
    cases that require direct RPC interaction.
</Callout>

### Return Value

The hook returns a `SolanaClient` object. You can use this object to make RPC calls to the Solana network.

### Example

This example demonstrates how to use the client to fetch the SOL balance of the connected account.

```tsx
import { useWalletUi, useWalletUiSolanaClient } from '@wallet-ui/react';
import { useState, useEffect } from 'react';

const LAMPORTS_PER_SOL = 1_000_000_000;

function AccountBalance() {
    const { account } = useWalletUi();
    const client = useWalletUiSolanaClient();
    const [balance, setBalance] = useState<number | null>(null);

    useEffect(() => {
        if (!account || !client) return;

        const getBalance = async () => {
            const { value } = await client.rpc.getBalance(account.address).send();
            setBalance(value / LAMPORTS_PER_SOL);
        };

        getBalance();
    }, [account, client]);

    if (!account) {
        return <p>Connect an account to see its balance.</p>;
    }

    return <p>Balance: {balance === null ? 'Loading...' : `${balance} SOL`}</p>;
}
```
